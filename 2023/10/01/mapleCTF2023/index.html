with open("map.txt", "r") as f:
    exec(f.read())

mp = list(map(lambda x : list(map(int, x.split(','))), mp))
mp.sort(key = lambda x : x[2])
fa = []
for i in range(1001):
    fa.append(i)

def find(x):
    if fa[x] == x:
        return x
    ret = find(fa[x])
    fa[x] = ret
    return ret

ans = 0
cnt = 0
_a = []
for a, b, res in mp:
    ffa = find(a)
    ffb = find(b)
    if (ffa == ffb):
        continue
    fa[ffa] = ffb
    ans += res
    cnt += 1
    _a.append(str(a) + ',' + str(b) + ',' + str(res))

with open("ans.txt", "w") as f:
    f.write(str(_a))
